- –í–µ–∫—Ç–æ—Ä —É–≥–ª–æ–≤–æ–π —Å–∫–æ—Ä–æ—Å—Ç–∏: $\vec{\vec\omega}=\frac{d\vec\omega}{dt}$

- **–í—Ä–∞—â–µ–Ω–∏–µ–º —Ç–≤–µ—Ä–¥–æ–≥–æ —Ç–µ–ª–∞** –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ –Ω–µ–ø–æ–¥–≤–∏–∂–Ω–æ–π —Ç–æ—á–∫–∏ –û (–ø–æ–ª—é—Å–∞) –Ω–∞–∑—ã–≤–∞—é—Ç –¥–≤–∏–∂–µ–Ω–∏–µ —Ç–≤–µ—Ä–¥–æ–≥–æ —Ç–µ–ª–∞, –ø—Ä–∏ –∫–æ—Ç–æ—Ä–æ–º —Ç–æ–ª—å–∫–æ –æ–¥–Ω–∞ –µ–≥–æ —Ç–æ—á–∫–∞ –û  –æ—Å—Ç–∞–µ—Ç—Å—è –≤—Å–µ –≤—Ä–µ–º—è –Ω–µ–ø–æ–¥–≤–∏–∂–Ω–æ–π, –∞ –æ—Å—Ç–∞–ª—å–Ω—ã–µ ‚Äì –¥–≤–∏–∂—É—Ç—Å—è –ø–æ –ø–æ–≤–µ—Ä—Ö–Ω–æ—Å—Ç—è–º –∫–æ–Ω—Ü–µ–Ω—Ç—Ä–∏—á–µ—Å–∫–∏—Ö —Å—Ñ–µ—Ä —Å —Ü–µ–Ω—Ç—Ä–æ–≤ –≤ –û. 
- –î–≤–∏–∂–µ–Ω–∏–µ –º–∞—Ç–µ—Ä–∏–∞–ª—å–Ω–æ–π —Ç–æ—á–∫–∏ –ø–æ –æ–∫—Ä—É–∂–Ω–æ—Å—Ç–∏ —Å –ø–æ—Å—Ç–æ—è–Ω–Ω—ã–º –º–æ–¥—É–ª–µ–º —Å–∫–æ—Ä–æ—Å—Ç–∏ ùúê=ùëêùëúùëõùë†ùë° —è–≤–ª—è–µ—Ç—Å—è —É—Å–∫–æ—Ä–µ–Ω–Ω—ã–º –¥–≤–∏–∂–µ–Ω–∏–µ–º (—Å —Ü–µ–Ω—Ç—Ä–æ—Å—Ç—Ä–µ–º–∏—Ç–µ–ª—å–Ω—ã–º —É—Å–∫–æ—Ä–µ–Ω–∏–µ–º $ùëé_{ùëõ}$). –í–µ–∫—Ç–æ—Ä –ø–æ–ª–Ω–æ–≥–æ —É—Å–∫–æ—Ä–µ–Ω–∏—è

$\vec\alpha=\overrightarrow{\alpha_{n}}+\vec{\alpha_{\tau}}$

# –í—Ä–∞—â–∞—Ç–µ–ª—å–Ω–æ–µ –¥–≤–∏–∂–µ–Ω–∏–µ

# –î–∏–Ω–∞–º–∏–∫–∞ –≤—Ä–∞—â–∞—Ç–µ–ª—å–Ω–æ–≥–æ –¥–≤–∏–∂–µ–Ω–∏—è –º–∞—Ç–µ—Ä–∏–∞–ª—å–Ω–æ–π —Ç–æ—á–∫–∏ –∏ —Ç–≤–µ—Ä–¥–æ–≥–æ —Ç–µ–ª–∞

    - –û—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ –Ω–µ–ø–æ–¥–≤–∏–∂–Ω–æ–π —Ç–æ—á–∫–∏ –û ‚Äì –≤–µ–∫—Ç–æ—Ä–Ω–∞—è –≤–µ–ª–∏—á–∏–Ω–∞.
    - –û—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ –æ—Å–∏ –ê–í ‚Äì —Å–∫–∞–ª—è—Ä–Ω–∞—è –≤–µ–ª–∏—á–∏–Ω–∞
- –û—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ —Ç–æ—á–∫–∏ –û

$$\vec{M}=[\vec{r}\cdot\vec{F}] = 
 \begin{vmatrix}

- –ú–æ–¥—É–ª—å –≤–µ–∫—Ç–æ—Ä–∞:

# –•–∞—Ä–∞–∫—Ç–µ—Ä–∏—Å—Ç–∏–∫–∏ –≤—Ä–∞—â–∞—Ç–µ–ª—å–Ω–æ–≥–æ –¥–≤–∏–∂–µ–Ω–∏—è

# –ú–æ–º–µ–Ω—Ç –∏–Ω–µ—Ä—Ü–∏–∏
- –ú–æ–º–µ–Ω—Ç –∏–Ω–µ—Ä—Ü–∏–∏ –º–∞—Ç–µ—Ä–∏–∞–ª—å–Ω–æ–π —Ç–æ—á–∫–∏ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ –Ω–µ–ø–æ–¥–≤–∏–∂–Ω–æ–π –æ—Å–∏ –ê–í —Ä–∞–≤–µ–Ω –ø—Ä–æ–∏–∑–≤–µ–¥–µ–Ω–∏—é –º–∞—Å—Å—ã –º–∞—Ç–µ—Ä–∏–∞–ª—å–Ω–æ–π —Ç–æ—á–∫–∏ –Ω–∞ –∫–≤–∞–¥—Ä–∞—Ç —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è —ç—Ç–æ–π —Ç–æ—á–∫–∏ –æ—Ç –æ—Å–∏ –≤—Ä–∞—â–µ–Ω–∏—è:

$J_{AB}=mr^2$

- –ú–æ–º–µ–Ω—Ç –∏–Ω–µ—Ä—Ü–∏–∏ –º–µ—Ö–∞–Ω–∏—á–µ—Å–∫–æ–π —Å–∏—Å—Ç–µ–º—ã, —Å–æ—Å—Ç–æ—è—â–µ–π –∏–∑ n –º–∞—Ç–µ—Ä–∏–∞–ª—å–Ω—ã—Ö —Ç–æ—á–µ–∫, –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ –Ω–µ–ø–æ–¥–≤–∏–∂–Ω–æ–π –æ—Å–∏ –ê–í:

$J_{AB}= \sum_{i=1}^{n}m_{i}r_{i}^2$

- –ú–æ–º–µ–Ω—Ç –∏–Ω–µ—Ä—Ü–∏–∏ —Ç–≤–µ—Ä–¥–æ–≥–æ —Ç–µ–ª–∞ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ –Ω–µ–ø–æ–¥–≤–∏–∂–Ω–æ–π –æ—Å–∏ –ê–í, —Å —É—á—ë—Ç–æ–º ùëëùëö=ùúå‚àôùëëùëâ: 

$J_{AB}=\int_m r^2dm=\rho\int_\nu r^2dV$

- **–ú–æ–º–µ–Ω—Ç –∏–Ω–µ—Ä—Ü–∏–∏ —Ç–≤–µ—Ä–¥–æ–≥–æ —Ç–µ–ª–∞** —è–≤–ª—è–µ—Ç—Å—è –º–µ—Ä–æ–π –∏–Ω–µ—Ä—Ü–∏–∏ —Ç–µ–ª–∞ –≤–æ –≤—Ä–∞—â–∞—Ç–µ–ª—å–Ω–æ–º –¥–≤–∏–∂–µ–Ω–∏–∏, –∫–∞–∫ –º–∞—Å—Å–∞ —Ç–µ–ª–∞ —è–≤–ª—è–µ—Ç—Å—è –º–µ—Ä–æ–π –µ–≥–æ –∏–Ω–µ—Ä—Ü–∏–∏ –≤ –ø–æ—Å—Ç—É–ø–∞—Ç–µ–ª—å–Ω–æ–º –¥–≤–∏–∂–µ–Ω–∏–∏.
- –ü—Ä–∏ –ø—Ä–æ–µ–∫—Ü–∏–∏ –º–æ–º–µ–Ω—Ç–∞ –∏–Ω–µ—Ä—Ü–∏–∏ –Ω–∞ –æ—Å—å z:
$J_{z}=\frac{m}{V}\int(y_k^2+x_k^2)dV$

# –ú–æ–º–µ–Ω—Ç –∏–Ω–µ—Ä—Ü–∏–∏
- –ù–∞–π–¥—ë–º –º–æ–º–µ–Ω—Ç –∏–Ω–µ—Ä—Ü–∏–∏ —Å—Ç–µ—Ä–∂–Ω—è, –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ —Ü–µ–Ω—Ç—Ä–∞–ª—å–Ω–æ–π –æ—Å–∏:

# –í—Ä–∞—â–∞—Ç–µ–ª—å–Ω–æ–µ –¥–≤–∏–∂–µ–Ω–∏–µ
- –ö–∏–Ω–µ—Ç–∏—á–µ—Å–∫–∞—è —ç–Ω–µ—Ä–≥–∏—è –ø—Ä–∏ –≤—Ä–∞—â–∞—Ç–µ–ª—å–Ω–æ–º –¥–≤–∏–∂–µ–Ω–∏–∏:
